// --- Main Gallery and Lightbox Logic --- function buildMainGallery() { imageUrls.forEach((url, index) => { const itemDiv = document.createElement('div'); itemDiv.className = 'gallery-item aspect-square'; const img = document.createElement('img'); img.src = url; img.alt = `Photo ${index + 1} by Danko Martinez`; img.loading = 'lazy'; itemDiv.onclick = () => openLightbox(index); itemDiv.appendChild(img); gallery.appendChild(itemDiv); }); } const showNextImage = () => { currentIndex = (currentIndex + 1) % imageUrls.length; lightboxImg.style.opacity = 0; setTimeout(() => { lightboxImg.src = imageUrls[currentIndex]; lightboxImg.style.opacity = 1; }, 100); // Match the very fast CSS transition }; const openLightbox = (index) => { currentIndex = index; lightboxImg.src = imageUrls[currentIndex]; lightbox.classList.add('visible'); slideshowInterval = setInterval(showNextImage, 300); // Change image every 0.3 seconds }; const closeLightbox = () => { lightbox.classList.remove('visible'); clearInterval(slideshowInterval); }; lightboxClose.onclick = closeLightbox; lightbox.onclick = (e) => { if (e.target === lightbox) closeLightbox(); }; // --- Execution Flow --- window.onload = () => { buildMainGallery(); };
