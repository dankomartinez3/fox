}; const closeLightbox = () => { lightbox.classList.remove('visible'); clearInterval(slideshowInterval); }; lightboxClose.onclick = closeLightbox; lightbox.onclick = (e) => { if (e.target === lightbox) closeLightbox(); }; // --- Execution Flow --- window.onload = () => { if (imageUrls.length > 0) { const pixelCount = createFoxLoader(); // Wait for loader animation to finish, then show main gallery setTimeout(() => { preloader.style.opacity = 0;
